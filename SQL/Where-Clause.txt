Starter SQL Queries (Only WHERE Clause)
---------AUTHORS Table
1. Find authors from a specific city
SELECT * 
FROM authors
WHERE city = 'New York';
2. Authors from a specific state
SELECT * 
FROM authors
WHERE state = 'CA';
3. Authors with contract = 1
SELECT * 
FROM authors
WHERE contract = 1;

----------TITLES Table
4. Books with price greater than 20
SELECT * 
FROM titles
WHERE price > 20;
5. Books of type 'business'
SELECT * 
FROM titles
WHERE type = 'business';
6. Books with advance between 5000 and 10000
SELECT * 
FROM titles
WHERE advance BETWEEN 5000 AND 10000;

-----------EMPLOYEES Table
